create database inventarioPersonalizado 
use inventarioPersonalizado

create table MA_Usuarios(
PK_idUsuario int unique auto_increment not null,
nombre varchar (50),
usuario varchar (15),
cedula int(10),
password varchar (20),
dependencia int not null,
cargo varchar (20) not null,
contrato varchar (20) not null,
);

create table MA_Bienes(
idBien int auto_increment not null,
PK_Codigo int(12) not null,
nombre varchar(50) not null,
placa varchar(6) not null,
descripcion varchar(150) not null,
valor int not null,
FK_Usuario int not null,
ubicacion varchar(50) not null,
imagen varchar(50) not null
);

create table PA_BienesPorUsuario(
+PK_idBienPorUsuario int, 
+FK_Usuario int not null,
+FK_Bien int,
+asunto varchar(25) not null,
+informacion varchar(200) not null,
+fechaObservacion date not null
);

CREATE PROCEDURE USP_informaciondebienMA_Bienes_I(
	IN p_idBien int,
	IN p_PK_Codigo int,
	IN p_placa int,
	IN p_nombre varchar(30),
	IN p_descripcion varchar(50),
	IN p_valor int,
	IN p_estado varchar(8),
	IN p_grupo int, 
	IN p_ubicacion int,
	IN p_imagen varchar(15),
	IN p_FK_Usuario int
)
BEGIN
    INSERT INTO MA_Bienes (idBien, PK_Codigo, placa, nombre, descripcion, valor, grupo, ubicacion, imagen, FK_Usuario)
    VALUES (p_idBien, p_PK_Codigo, p_placa, p_nombre, p_descripcion, p_valor, p_grupo, p_ubicacion, p_imagen, p_FK_Usuario);
END //
DELIMITER;

DELIMITER //
CREATE PROCEDURE USP_estadobienMA_Bienes_U(
	IN p_PK_idBien int,
	IN p_estado varchar(8)
)
BEGIN
    UPDATE INTO MA_Bienes (PK_idBien, estado)
    VALUES (p_PK_idBien, p_estado);
END //
DELIMITER;


--- Autor: Cesar Salazar
Fecha creación: 18/03/2024
Descripción de los parámetros de entrada: Los parametros del procedimiento son las columna PK_idBien la cual identificara el procedimiento y la columna estado la cual sera la que sufrira el cambio, este cambio sera de Revisado, Correcto o Espera en la tabla MA_BienesPorUsuario.
Descripción de los parámetros de salida: Los valores que se muestran en la salida sera el cambio de estado del bien el cual iniciara con un campo nulo y al ejecutar el procedimiento el estado quedara con el valor sea el caso.
Objetivo del procedimiento almacenado/función: El objetivo del procedimiento es actualizar el campo estado en la tabla MA_Bienes.
